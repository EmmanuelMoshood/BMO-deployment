
###clone and setup my wordpress app for bmo

#update apt package
sudo apt-get update

#Install Apache server on Ubuntu
sudo apt install apache2 -y

#Install php runtime and php mysql connector
sudo apt install php libapache2-mod-php php-mysql -y




#Install MySQL server
sudo apt install mysql-server -y

#Login to MySQL server
sudo mysql -u root

#Change authentication plugin to mysql_native_password (change the password to something strong)
ALTER USER 'root'@'localhost' IDENTIFIED WITH mysql_native_password by 'Testpassword@123';

#Create a new database user for wordpress (change the password to something strong)
CREATE USER 'wp_user'@localhost IDENTIFIED BY 'Testpassword@123';

#Create a database for wordpress
CREATE DATABASE wp;

#Grant all privilges on the database 'wp' to the newly created user
GRANT ALL PRIVILEGES ON wp.* TO 'wp_user'@localhost;




#Clone wp app to /tmp
cd /tmp
git clone https://github.com/EmmanuelMoshood/BMO-deployment.git

#Move BMO-deployment folder to apache document root
sudo mv BMO-deployment/ /var/www/html

#Command to restart/reload apache server
sudo systemctl restart apache2
#or sudo systemctl reload apache2
